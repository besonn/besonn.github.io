龙贝格算法和梯形算法求解积分的比较

2020-04-20
#include <cstdio>
#include <stdlib.h>
#include <math.h>
int Rk = 0; // 龙贝格二分次数
int Tk = 0; // 梯形二分次数
double c2[4] = { -1 }, t2[4] = { -1 }, s2[4] = { -1 }; // 分别用来保存龙贝格中间结果
double fx(double x)
{
	return x*x*x-x*x+1; // 自由设置函数
}
double Romberg(double a, double b, double e)
{
	// 返回龙贝格结果
	int cnt = 1;
	double T1, T2, S1, S2, C1, R1,h,S,x;
	double C2 = 0, R2 = 0;
	h = b - a;
	int k = 1;
	T1 = h / 2.0*(fx(a) + fx(b));
	while (true)
	{
		Rk++;
		cnt++;
		S = 0;
		x = a + h / 2.0;
		int cnt = 0;
		do
		{
			S = S + fx(x);
			x = x + h;
		} while (x < b);

		T2 = T1 / 2.0 + h / 2.0*S;
		t2[Rk-1] = T2;
		S2 = T2 + (T2 - T1) / 3.0;
		s2[Rk-1] = S2;
		h = h / 2.0;
		T1 = T2;
		S1 = S2;
		C1 = C2;
		R1 = R2;
		if (k == 1)
		{
			k++;
			continue;
		}
		C2 = S2 + (S2 - S1) / 15.0;
		c2[Rk - 1] = C2;
		if (k == 2)
		{
			k++;
			continue;
		}
		R2 = C2 + (C2 - C1) / 63.0;
		if (k == 3)
		{
			k++;
			continue;
		}
		if (fabs(R1 - R2) < e || cnt >= 100)
		{
			break;
		}
	}
	return R2;
}
double Tn(double a, double b, double e)
{
	// 返回梯形结果
	    double T1=0, T2=0;
	    double h = b - a;
	    T1 = (fx(a) + fx(b))*h / 2.0;
	    while (1)
		{
		         Tk++;
				 double s = 0.0;
				 for (double i = a + h / 2.0; i < b; i += h) 
				 {
				       s += fx(i);
				 }
		         T2 = (T1 + h * s) / 2.0;
		         if (fabs(T2 - T1) < e)break;
		         h /= 2.0;
		         T1 = T2;
		}
	     return T2;
	 }
int main()
{
	double a, b, e;
	printf("请输入a,b,e：");
	scanf("%lf%lf%lf", &a, &b, &e);
	double r = Romberg(a, b, e);
	printf("\nRomberg:积分值：%.7lf  --  二分次数：%d\n", r, Rk);
	// 输出中间结果
	printf("%lf %lf %lf %lf\n", t2[0], t2[1], t2[2], t2[3]);
	printf("%lf %lf %lf %lf\n", s2[0], s2[1], s2[2], s2[3]);
	printf("%lf %lf %lf %lf\n", c2[0], c2[1], c2[2], c2[3]);
	double t = Tn(a, b, e);
    printf("\nTn:积分值：%.7lf  --  二分次数：%d\n", t, Tk);
}
